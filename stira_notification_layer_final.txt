HOW THIS WORKS
This document builds a complete on-device notification intelligence system using flutter_local_notifications + timezone. Zero Firebase Cloud Functions. Zero Blaze plan. The system runs a background engine every 15 minutes using workmanager, reads local Hive storage, analyses the user's check-in history, location, streak, and danger zones, then schedules or cancels local notifications in real time. Every notification fires from the device itself ΓÇö no server needed, ever.
Architecture Overview
Six files work together as one system. Each has a single responsibility. They communicate through Hive (local storage) ΓÇö no network calls between them.

import 'package:timezone/timezone.dart' as tz;
import 'package:timezone/data/latest.dart' as tz;
import 'package:vibration/vibration.dart';

class StiraNotificationService {
  static final FlutterLocalNotificationsPlugin _plugin =
      FlutterLocalNotificationsPlugin();
  static bool _initialized = false;
  // ΓöÇΓöÇ Channel IDs ΓöÇΓöÇ
  static const String CHANNEL_CRITICAL  = 'stira_critical';
  static const String CHANNEL_HIGH      = 'stira_high';
  static const String CHANNEL_NORMAL    = 'stira_normal';
  static const String CHANNEL_SILENT    = 'stira_silent';

  static const int ID_CHECKIN_REMINDER   = 1001;

  static Future<void> initialize() async {
    if (_initialized) return;
    tz.initializeTimeZones();
    // Android init
    const AndroidInitializationSettings androidSettings =
        AndroidInitializationSettings("@mipmap/ic_launcher");
    // iOS init
    const DarwinInitializationSettings iosSettings =
        DarwinInitializationSettings(
      requestAlertPermission: false,   // we request manually below
      requestBadgePermission: true,
      requestSoundPermission: true,
    );
    const InitializationSettings settings = InitializationSettings(
      android: androidSettings,
      iOS: iosSettings,
    );
    await _plugin.initialize(
      settings,
      onDidReceiveNotificationResponse: _onNotificationTapped,
      onDidReceiveBackgroundNotificationResponse: _onBackgroundNotificationTapped,
    );
    await _createChannels();
    _initialized = true;
  }
  static Future<void> requestPermission() async {
    if (await Permission.notification.isDenied) {
      await Permission.notification.request();
    }
    // iOS specific
    await _plugin
        .resolvePlatformSpecificImplementation<
            IOSFlutterLocalNotificationsPlugin>()
        ?.requestPermissions(alert: true, badge: true, sound: true);
  }
  static Future<void> _createChannels() async {
    final androidPlugin = _plugin.resolvePlatformSpecificImplementation<
        AndroidFlutterLocalNotificationsPlugin>();
    if (androidPlugin == null) return;
    // CRITICAL: Danger zones, crisis ΓÇö max importance, always vibrate
    await androidPlugin.createNotificationChannel(
      const AndroidNotificationChannel(
        'stira_critical',
        'Stira Critical Alerts',
        description: 'Danger zones and crisis moments',
        importance: Importance.max,
        enableVibration: true,
        playSound: true,
        showBadge: true,
      ),
    );
    // HIGH: Milestones, vulnerability warnings, check-in reminders
    await androidPlugin.createNotificationChannel(
      const AndroidNotificationChannel(
        'stira_high',
        'Stira Alerts',
        description: 'Vulnerability warnings and check-in reminders',
        importance: Importance.high,
        enableVibration: true,
        playSound: true,
        showBadge: true,
      ),
    );
    // NORMAL: Journal reminders, identity, weekly report
    await androidPlugin.createNotificationChannel(
      const AndroidNotificationChannel(
        'stira_normal',
        'Stira Reminders',
        description: 'Journal reminders and daily check-ins',
        importance: Importance.defaultImportance,
        enableVibration: false,
        playSound: true,
        showBadge: false,
      ),
    );
  }
  // ΓöÇΓöÇ SCHEDULE a notification at a specific future time ΓöÇΓöÇ
  static Future<void> scheduleAt({
    required int id,
    required String title,
    required String body,
    required DateTime scheduledTime,
    String channel = CHANNEL_HIGH,
    String? payload,
  }) async {
    if (!_initialized) await initialize();
    final tz.TZDateTime tzTime = tz.TZDateTime.from(
      scheduledTime,
      tz.local,
    );
    // Cancel existing notification with same ID first
    await _plugin.cancel(id);
    await _plugin.zonedSchedule(
      id,
      title,
      body,
      tzTime,
      NotificationDetails(
        android: AndroidNotificationDetails(
          channel,
          _channelName(channel),
          importance: _channelImportance(channel),
          priority: Priority.high,
          enableVibration: channel != CHANNEL_NORMAL,
          styleInformation: BigTextStyleInformation(body),
          icon: "@mipmap/ic_launcher",
        ),
        iOS: const DarwinNotificationDetails(
          presentAlert: true,
          presentBadge: true,
          presentSound: true,
        ),
      ),
      androidScheduleMode: AndroidScheduleMode.exactAllowWhileIdle,
      uiLocalNotificationDateInterpretation:
          UILocalNotificationDateInterpretation.absoluteTime,
      payload: payload,
    );
  }
  // ΓöÇΓöÇ SHOW immediately (for Danger Zone, crisis) ΓöÇΓöÇ
  static Future<void> showNow({
    required int id,
    required String title,
    required String body,
    String channel = CHANNEL_CRITICAL,
    String? payload,
  }) async {
    if (!_initialized) await initialize();
    await _plugin.cancel(id);
    await _plugin.show(
      id,
      title,
      body,
      NotificationDetails(
        android: AndroidNotificationDetails(
          channel,
          _channelName(channel),
          importance: Importance.max,
          priority: Priority.max,
          enableVibration: true,
          styleInformation: BigTextStyleInformation(body),
        ),
        iOS: const DarwinNotificationDetails(
          presentAlert: true, presentBadge: true, presentSound: true,
        ),
      ),
      payload: payload,
    );
  }
  // ΓöÇΓöÇ SOS HEARTBEAT ΓÇö haptic only, no visual notification ΓöÇΓöÇ
  static Future<void> fireSOSHeartbeat() async {
    if (await Vibration.hasVibrator() ?? false) {
      // Pattern: 3 short ΓåÆ pause ΓåÆ 1 long ΓåÆ pause ΓåÆ 3 short
      await Vibration.vibrate(
        pattern: [0, 100, 100, 100, 100, 100, 200, 600, 200, 100, 100, 100, 100, 100],
        intensities: [0, 255, 0, 255, 0, 255, 0, 255, 0, 255, 0, 255, 0, 255],
      );
    }
  }
  // ΓöÇΓöÇ CANCEL a specific notification ΓöÇΓöÇ
  static Future<void> cancel(int id) async => _plugin.cancel(id);
  // ΓöÇΓöÇ CANCEL ALL scheduled notifications ΓöÇΓöÇ
  static Future<void> cancelAll() async => _plugin.cancelAll();
  // ΓöÇΓöÇ TAP HANDLER ΓÇö deep link routing ΓöÇΓöÇ
  static void _onNotificationTapped(NotificationResponse response) {
    final payload = response.payload ?? "";
    // Route based on payload ΓÇö handled by NavigationService
    StiraNavigationService.handleNotificationTap(payload);
  }
  @pragma("vm:entry-point")
  static void _onBackgroundNotificationTapped(NotificationResponse response) {
    // Store payload in Hive so app reads it on next open
    final box = Hive.box("notification_taps");
    box.put("pending_tap", response.payload);
  }
  static String _channelName(String id) {
    if (id == CHANNEL_CRITICAL) return 'Stira Critical Alerts';
    if (id == CHANNEL_HIGH) return 'Stira Alerts';
    if (id == CHANNEL_NORMAL) return 'Stira Reminders';
    return 'Stira';
  }
  static Importance _channelImportance(String id) {
    if (id == CHANNEL_CRITICAL) return Importance.max;
    if (id == CHANNEL_HIGH) return Importance.high;
    return Importance.defaultImportance;
  }
}

// FILE: lib/services/stira_behaviour_analyser.dart
import 'package:hive/hive.dart';
// Output model ΓÇö all signals in one object
class BehaviourSnapshot {
  final int currentStreak;
  final int hoursSinceLastCheckIn;
  final double avgUrgeIntensityLast7Days;
  final double urgeVelocity;          // positive = rising, negative = falling
  final int peakHour;                 // 0-23: hour with most high-intensity check-ins
  final int peakHourConfidence;       // 0-100: how confident the peak hour prediction is
  final String topTrigger;            // "Boredom", "Stress", etc.
  final bool checkedInToday;
  final bool dopamineJournalToday;
  final bool isInRestMode;
  final int sentToday;
  final DateTime? lastSentAt;
  final double stabilityTrend;        // positive = improving, negative = declining
  final int totalCleanDays;
  final String userName;
  final String coreNeed;
  final int daysSinceInstall;
  final bool sameTimeTomorrowDue;
  final String sameTimeTomorrowCommitment;
  final DateTime? lastRelapseAt;
  const BehaviourSnapshot({
    required this.currentStreak,
    required this.hoursSinceLastCheckIn,
    required this.avgUrgeIntensityLast7Days,
    required this.urgeVelocity,
    required this.peakHour,
    required this.peakHourConfidence,
    required this.topTrigger,
    required this.checkedInToday,
    required this.dopamineJournalToday,
    required this.isInRestMode,
    required this.sentToday,
    required this.lastSentAt,
    required this.stabilityTrend,
    required this.totalCleanDays,
    required this.userName,
    required this.coreNeed,
    required this.daysSinceInstall,
    required this.sameTimeTomorrowDue,
    required this.sameTimeTomorrowCommitment,
    required this.lastRelapseAt,
  });
}
class StiraBehaviourAnalyser {
  static BehaviourSnapshot analyse() {
    final checkIns    = Hive.box('check_ins');
    final userData    = Hive.box('user_data');
    final notifState  = Hive.box('notification_state');
    final journal     = Hive.box('dopamine_journal');
    final commitments = Hive.box('commitments');
    // ΓöÇΓöÇ Basic user data ΓöÇΓöÇ
    final String userName   = userData.get('name', defaultValue: 'there');
    final String coreNeed   = userData.get('core_need', defaultValue: 'Calm');
    final int currentStreak = userData.get('current_streak', defaultValue: 0);
    final int totalCleanDays= userData.get('total_clean_days', defaultValue: 0);
    final DateTime installDate = DateTime.parse(userData.get('install_date',
        defaultValue: DateTime.now().toIso8601String()));
    final int daysSinceInstall = DateTime.now().difference(installDate).inDays;
    // ΓöÇΓöÇ Last relapse ΓöÇΓöÇ
    final String? lastRelapseStr = userData.get('last_relapse_at');
    final DateTime? lastRelapseAt = lastRelapseStr != null
        ? DateTime.tryParse(lastRelapseStr) : null;
    // ΓöÇΓöÇ Check-in history (last 30 days) ΓöÇΓöÇ
    final now = DateTime.now();
    final allCheckIns = checkIns.values
        .cast<Map>().toList()
        .where((c) {
          final ts = DateTime.tryParse(c['timestamp'] ?? '');
          return ts != null && now.difference(ts).inDays <= 30;
        })
        .toList()
        ..sort((a, b) {
          final ta = DateTime.parse(a['timestamp']);
          final tb = DateTime.parse(b['timestamp']);
          return tb.compareTo(ta); // newest first
        });
    // ΓöÇΓöÇ Hours since last check-in ΓöÇΓöÇ
    int hoursSinceLastCheckIn = 999;
    if (allCheckIns.isNotEmpty) {
      final last = DateTime.parse(allCheckIns.first['timestamp']);
      hoursSinceLastCheckIn = now.difference(last).inHours;
    }
    // ΓöÇΓöÇ Checked in today ΓöÇΓöÇ
    final bool checkedInToday = allCheckIns.any((c) {
      final ts = DateTime.parse(c['timestamp']);
      return ts.day == now.day && ts.month == now.month && ts.year == now.year;
    });
    // ΓöÇΓöÇ Average urge intensity last 7 days ΓöÇΓöÇ
    final last7 = allCheckIns.where((c) {
      final ts = DateTime.parse(c['timestamp']);
      return now.difference(ts).inDays <= 7;
    }).toList();
    final double avgUrge = last7.isEmpty ? 0 :
        last7.map((c) => (c['urge_level'] as num).toDouble())
        .reduce((a, b) => a + b) / last7.length;
    // ΓöÇΓöÇ Urge velocity: compare last 3 check-ins ΓöÇΓöÇ
    double urgeVelocity = 0;
    if (allCheckIns.length >= 3) {
      final levels = allCheckIns.take(3).map((c) => (c['urge_level'] as num).toDouble()).toList();
      urgeVelocity = levels[0] - levels[2]; // newest minus oldest of 3
    }
    // ΓöÇΓöÇ Peak hour: find the hour with most high-intensity (>=6) check-ins ΓöÇΓöÇ
    final Map<int, int> hourCounts = {};
    for (final c in allCheckIns) {
      if ((c['urge_level'] as num) >= 6) {
        final hour = DateTime.parse(c['timestamp']).hour;
        hourCounts[hour] = (hourCounts[hour] ?? 0) + 1;
      }
    }
    int peakHour = 21; // default: 9 PM
    int peakCount = 0;
    hourCounts.forEach((h, count) {
      if (count > peakCount) { peakCount = count; peakHour = h; }
    });
    // Confidence: how many data points support this peak
    final int peakHourConfidence = (peakCount / (last7.length + 1) * 100).clamp(0, 100).toInt();
    // ΓöÇΓöÇ Top trigger ΓöÇΓöÇ
    final Map<String, int> triggerCounts = {};
    for (final c in allCheckIns) {
      final triggers = List<String>.from(c['triggers'] ?? []);
      for (final t in triggers) {
        triggerCounts[t] = (triggerCounts[t] ?? 0) + 1;
      }
    }
    String topTrigger = 'Stress';
    int topCount = 0;
    triggerCounts.forEach((t, count) {
      if (count > topCount) { topCount = count; topTrigger = t; }
    });
    // ΓöÇΓöÇ Stability trend: compare avg intensity week1 vs week2 ΓöÇΓöÇ
    final week1 = allCheckIns.where((c) {
      final d = now.difference(DateTime.parse(c['timestamp'])).inDays;
      return d <= 7;
    }).toList();
    final week2 = allCheckIns.where((c) {
      final d = now.difference(DateTime.parse(c['timestamp'])).inDays;
      return d > 7 && d <= 14;
    }).toList();
    double stabilityTrend = 0;
    if (week1.isNotEmpty && week2.isNotEmpty) {
      final avg1 = week1.map((c) => (c['urge_level'] as num).toDouble()).reduce((a,b)=>a+b)/week1.length;
      final avg2 = week2.map((c) => (c['urge_level'] as num).toDouble()).reduce((a,b)=>a+b)/week2.length;
      stabilityTrend = avg2 - avg1; // positive = getting worse, negative = improving
    }
    // ΓöÇΓöÇ Dopamine journal today ΓöÇΓöÇ
    final bool dopamineJournalToday = journal.values.any((j) {
      final ts = DateTime.tryParse(j['date'] ?? '');
      return ts != null && ts.day == now.day && ts.month == now.month;
    });
    // ΓöÇΓöÇ Notification state ΓöÇΓöÇ
    final int sentToday = notifState.get('sent_today', defaultValue: 0);
    final bool isInRestMode = notifState.get('rest_mode', defaultValue: false);
    final String? lastSentStr = notifState.get('last_sent_at');
    final DateTime? lastSentAt = lastSentStr != null ? DateTime.tryParse(lastSentStr) : null;
    // ΓöÇΓöÇ Same Time Tomorrow commitment ΓöÇΓöÇ
    bool sameTimeTomorrowDue = false;
    String sameTimeTomorrowCommitment = '';
    if (commitments.isNotEmpty) {
      final recent = commitments.values.last as Map?;
      if (recent != null) {
        final savedAt = DateTime.tryParse(recent['saved_at'] ?? '');
        if (savedAt != null) {
          final hoursDiff = now.difference(savedAt).inHours;
          if (hoursDiff >= 23 && hoursDiff <= 25 && recent['outcome'] == null) {
            sameTimeTomorrowDue = true;
            sameTimeTomorrowCommitment = recent['commitment'] ?? '';
          }
        }
      }
    }
    return BehaviourSnapshot(
      currentStreak: currentStreak,
      hoursSinceLastCheckIn: hoursSinceLastCheckIn,
      avgUrgeIntensityLast7Days: avgUrge,
      urgeVelocity: urgeVelocity,
      peakHour: peakHour,
      peakHourConfidence: peakHourConfidence,
      topTrigger: topTrigger,
      checkedInToday: checkedInToday,
      dopamineJournalToday: dopamineJournalToday,
      isInRestMode: isInRestMode,
      sentToday: sentToday,
      lastSentAt: lastSentAt,
      stabilityTrend: stabilityTrend,
      totalCleanDays: totalCleanDays,
      userName: userName,
      coreNeed: coreNeed,
      daysSinceInstall: daysSinceInstall,
      sameTimeTomorrowDue: sameTimeTomorrowDue,
      sameTimeTomorrowCommitment: sameTimeTomorrowCommitment,
      lastRelapseAt: lastRelapseAt,
    );
  }
  // ΓöÇΓöÇ REAL-TIME: call this whenever the user does something in the app ΓöÇΓöÇ
  // Pass the snapshot to IntelligenceEngine.reactToUserAction()
  static void syncToHive({
    required int currentStreak,
    required int urgeLevel,
    required List<String> triggers,
    required String location,
  }) {

    checkIns.add({

      'location': location,
    });

    userData.put('current_streak', currentStreak);
    userData.put('last_checkin_at', DateTime.now().toIso8601String());
  }
}

// FILE: lib/services/stira_throttle_manager.dart
import 'package:hive/hive.dart';
class StiraThrottleManager {
  static const int NORMAL_DAILY_LIMIT  = 2;
  static const int CRISIS_DAILY_LIMIT  = 4;
  static const int REST_MODE_LIMIT      = 1;
  static const int MIN_GAP_HOURS        = 3;
  static const int QUIET_START_HOUR     = 22; // 10 PM
  static const int QUIET_END_HOUR       = 8;  // 8 AM
  // Returns true if a notification CAN be sent right now
  static bool canSend({bool isCrisis = false}) {
    final box = Hive.box('notification_state');
    final int sentToday = box.get('sent_today', defaultValue: 0);
    final bool restMode = box.get('rest_mode', defaultValue: false);
    final String? lastSentStr = box.get('last_sent_at');
    final bool crisisActive = box.get('crisis_active', defaultValue: false);
    // Reset sent_today at midnight
    final String? lastResetStr = box.get('last_reset_date');
    final today = DateTime.now().toLocal();
    final todayStr = "${today.year}-${today.month}-${today.day}";
    if (lastResetStr != todayStr) {
      box.put('sent_today', 0);
      box.put('last_reset_date', todayStr);
      // Also reset untapped counter each day
    }
    // Check quiet hours
    final hour = today.hour;
    if (hour >= QUIET_START_HOUR || hour < QUIET_END_HOUR) {
      if (!isCrisis) return false; // crisis overrides quiet hours
    }
    // Check rest mode
    if (restMode && sentToday >= REST_MODE_LIMIT) return false;
    // Check daily limit
    final limit = restMode ? REST_MODE_LIMIT
        : (crisisActive ? CRISIS_DAILY_LIMIT : NORMAL_DAILY_LIMIT);
    if (sentToday >= limit) return false;
    // Check minimum gap (3 hours between notifications)
    if (lastSentStr != null) {
      final lastSent = DateTime.tryParse(lastSentStr);
      if (lastSent != null) {
        final gap = DateTime.now().difference(lastSent).inHours;
        if (gap < MIN_GAP_HOURS && !isCrisis) return false;
      }
    }
    return true;
  }
  // Call this after every notification is sent
  static void recordSent() {
    final box = Hive.box('notification_state');
    final int sentToday = box.get('sent_today', defaultValue: 0);
    box.put('sent_today', sentToday + 1);
    box.put('last_sent_at', DateTime.now().toIso8601String());
  }
  // Call this when a notification is tapped
  static void recordTap() {
    final box = Hive.box('notification_state');
    box.put('consecutive_untapped', 0);
    // Exit rest mode if user is engaging again
    if (box.get('rest_mode', defaultValue: false)) {
      box.put('rest_mode', false);
    }
  }
  // Call this when a notification is NOT tapped (check at next cycle)
  static void recordUntapped() {
    final box = Hive.box('notification_state');
    final int untapped = box.get('consecutive_untapped', defaultValue: 0) + 1;
    box.put('consecutive_untapped', untapped);
    // After 5 consecutive untapped ΓåÆ rest mode for 7 days
    if (untapped >= 5) {
      box.put('rest_mode', true);
      box.put('rest_mode_until',
          DateTime.now().add(const Duration(days: 7)).toIso8601String());
    }
  }
  static void setCrisisMode(bool active) {
    Hive.box('notification_state').put('crisis_active', active);
  }
  // Check if rest mode period has expired
  static void checkRestModeExpiry() {
    final box = Hive.box('notification_state');
    final untilStr = box.get('rest_mode_until');
    if (untilStr != null) {
      final until = DateTime.tryParse(untilStr);
      if (until != null && DateTime.now().isAfter(until)) {
        box.put('rest_mode', false);
        box.put('consecutive_untapped', 0);
      }
    }
  }
}

The main decision engine ΓÇö runs every 15 minutes via workmanager, reacts in real-time to user actions
This is the core of the system. It runs on a 15-minute background cycle AND is called directly from inside the app whenever the user does something significant (checks in, logs a relapse, saves a commitment). That is what makes it real-time.

// ΓöÇΓöÇ BACKGROUND ENTRY POINT (must be top-level function) ΓöÇΓöÇ
@pragma("vm:entry-point")
void callbackDispatcher() {
  Workmanager().executeTask((task, inputData) async {
    if (task == TASK_INTELLIGENCE_CYCLE) {
      await Hive.initFlutter();
      await Hive.openBox('check_ins');
      await Hive.openBox('user_data');
      await Hive.openBox('notification_state');
      await Hive.openBox('dopamine_journal');

      await Hive.openBox('danger_zones');
      await Hive.openBox('notification_taps');
      await StiraNotificationService.initialize();
      await StiraIntelligenceEngine.runCycle();
    }
    return Future.value(true);
  });
}
class StiraIntelligenceEngine {
  // ΓöÇΓöÇ CALL FROM main.dart ONCE ON STARTUP ΓöÇΓöÇ
  static Future<void> startBackgroundCycle() async {
    await Workmanager().initialize(
      callbackDispatcher,
      isInDebugMode: false,
    );
    await Workmanager().registerPeriodicTask(
      'stira_intelligence_task',
      TASK_INTELLIGENCE_CYCLE,
      frequency: const Duration(minutes: 15),
      constraints: Constraints(
        networkType: NetworkType.not_required,
        requiresBatteryNotLow: false,
        requiresCharging: false,
        requiresDeviceIdle: false,
        requiresStorageNotLow: false,
      ),
      existingWorkPolicy: ExistingWorkPolicy.keep,
    );
  }
  // ΓöÇΓöÇ MAIN CYCLE ΓÇö called every 15 minutes ΓöÇΓöÇ
  static Future<void> runCycle() async {
    StiraThrottleManager.checkRestModeExpiry();
    final s = StiraBehaviourAnalyser.analyse();
    final now = DateTime.now();
    // ΓöÇΓöÇ STEP 1: CRISIS SCAN ΓÇö highest priority, ignores throttle ΓöÇΓöÇ
    if (_shouldFireUrgeVelocity(s)) {
      await _send(
        id: StiraNotificationService.ID_URGE_VELOCITY,
        content: StiraNotificationCopy.urgeVelocity(s),
        channel: StiraNotificationService.CHANNEL_CRITICAL,
        isCrisis: true,
      );
      return; // One notification per cycle. Stop here.
    }
    // ΓöÇΓöÇ STEP 2: SAME TIME TOMORROW ΓÇö highest engagement rate ΓöÇΓöÇ
    if (s.sameTimeTomorrowDue) {
      if (StiraThrottleManager.canSend()) {
        await _send(
          id: StiraNotificationService.ID_SAME_TIME_TOMORROW,
          content: StiraNotificationCopy.sameTimeTomorrow(
              s.sameTimeTomorrowCommitment, s.userName),
          channel: StiraNotificationService.CHANNEL_HIGH,
        );
        return;
      }
    }
    // ΓöÇΓöÇ STEP 3: MILESTONE SCAN ΓÇö never throttled ΓöÇΓöÇ
    final milestoneDays = [3, 7, 14, 30, 60, 90, 180, 365];
    if (milestoneDays.contains(s.currentStreak)) {
      final lastMilestoneDay = Hive.box('notification_state').get('last_milestone_day', defaultValue: -1);
      if (lastMilestoneDay != s.currentStreak) {
        Hive.box('notification_state').put('last_milestone_day', s.currentStreak);
        await _send(
          id: StiraNotificationService.ID_STREAK_MILESTONE,
          content: StiraNotificationCopy.streakMilestone(s.currentStreak, s.userName),
          channel: StiraNotificationService.CHANNEL_HIGH,
          isCrisis: true, // never throttled
        );
        return;
      }
    }
    // ΓöÇΓöÇ STEP 4: PRE-MILESTONE (tomorrow) ΓöÇΓöÇ
    final tomorrow = s.currentStreak + 1;
    if (milestoneDays.contains(tomorrow) && now.hour == 20) {
      if (StiraThrottleManager.canSend()) {
        await _send(
          id: StiraNotificationService.ID_PRE_MILESTONE,
          content: StiraNotificationCopy.preMilestone(tomorrow),
          channel: StiraNotificationService.CHANNEL_HIGH,
        );
        return;
      }
    }
    // ΓöÇΓöÇ STEP 5: PREDICTIVE VULNERABILITY WARNING ΓöÇΓöÇ
    if (_isApproachingPeakWindow(s, now) && StiraThrottleManager.canSend()) {
      await _send(
        id: StiraNotificationService.ID_VULNERABILITY,
        content: StiraNotificationCopy.vulnerabilityWarning(s),
        channel: StiraNotificationService.CHANNEL_HIGH,
      );
      return;
    }
    // ΓöÇΓöÇ STEP 6: CHECK-IN REMINDER ΓöÇΓöÇ
    if (_shouldFireCheckInReminder(s, now) && StiraThrottleManager.canSend()) {
      await _send(
        id: StiraNotificationService.ID_CHECKIN_REMINDER,
        content: StiraNotificationCopy.checkInReminder(s),
        channel: StiraNotificationService.CHANNEL_HIGH,
      );
      return;
    }
    // ΓöÇΓöÇ STEP 7: DISENGAGEMENT ΓöÇΓöÇ
    if (s.hoursSinceLastCheckIn >= 72 && StiraThrottleManager.canSend()) {
      await _send(
        id: StiraNotificationService.ID_DISENGAGEMENT_72H,
        content: StiraNotificationCopy.disengagement72h(s),
        channel: StiraNotificationService.CHANNEL_NORMAL,
      );
      return;
    }
    if (s.hoursSinceLastCheckIn >= 48 && StiraThrottleManager.canSend()) {
      await _send(
        id: StiraNotificationService.ID_DISENGAGEMENT_48H,
        content: StiraNotificationCopy.disengagement48h(s),
        channel: StiraNotificationService.CHANNEL_NORMAL,
      );
      return;
    }
    // ΓöÇΓöÇ STEP 8: DOPAMINE JOURNAL REMINDER (9 PM if no entry today) ΓöÇΓöÇ
    if (now.hour == 21 && !s.dopamineJournalToday && StiraThrottleManager.canSend()) {
      await _send(
        id: StiraNotificationService.ID_DOPAMINE_JOURNAL,
        content: StiraNotificationCopy.dopamineJournal(),
        channel: StiraNotificationService.CHANNEL_NORMAL,
      );
      return;
    }
    // ΓöÇΓöÇ STEP 9: IDENTITY PUSH ΓÇö only if nothing else fired today ΓöÇΓöÇ
    if (s.sentToday == 0 && now.hour == s.peakHour && StiraThrottleManager.canSend()) {
      await _send(
        id: StiraNotificationService.ID_IDENTITY_PUSH,
        content: StiraNotificationCopy.identityPush(s.daysSinceInstall),
        channel: StiraNotificationService.CHANNEL_NORMAL,
      );
      return;
    }
    // ΓöÇΓöÇ STEP 10: WEEKLY REPORT ΓÇö Sunday 8 AM ΓöÇΓöÇ
    if (now.weekday == DateTime.sunday && now.hour == 8 && StiraThrottleManager.canSend()) {
      await _send(
        id: StiraNotificationService.ID_WEEKLY_REPORT,
        content: StiraNotificationCopy.weeklyReport(),
        channel: StiraNotificationService.CHANNEL_NORMAL,
      );
      return;
    }
  }
  // ΓöÇΓöÇ REAL-TIME REACTION ΓÇö call immediately from UI after user actions ΓöÇΓöÇ
  static Future<void> reactToUserAction(UserAction action) async {
    await StiraNotificationService.initialize();
    final s = StiraBehaviourAnalyser.analyse();
    switch (action) {
      case UserAction.checkInSubmitted:
        // Cancel pending check-in reminder ΓÇö they already checked in
        await StiraNotificationService.cancel(StiraNotificationService.ID_CHECKIN_REMINDER);
        await StiraNotificationService.cancel(StiraNotificationService.ID_DISENGAGEMENT_48H);
        await StiraNotificationService.cancel(StiraNotificationService.ID_DISENGAGEMENT_72H);
        // If intensity >= 8: crisis mode + SOS heartbeat
        final lastCheckIn = Hive.box('check_ins').values.last as Map?;
        final urgeLevel = (lastCheckIn?['urge_level'] as num?)?.toInt() ?? 0;
        if (urgeLevel >= 8) {
          StiraThrottleManager.setCrisisMode(true);
          await StiraNotificationService.fireSOSHeartbeat();
        } else {
          StiraThrottleManager.setCrisisMode(false);
        }
        // Schedule a follow-up check-in reminder in 4 hours
        await StiraNotificationService.scheduleAt(
          id: StiraNotificationService.ID_POST_CHECKIN,
          title: StiraNotificationCopy.postCheckInFollowUp(s, 4).title,
          body: StiraNotificationCopy.postCheckInFollowUp(s, 4).body,
          scheduledTime: DateTime.now().add(const Duration(hours: 4)),
          channel: StiraNotificationService.CHANNEL_NORMAL,
          payload: "/checkin",
        );
        break;
      case UserAction.relapseLogged:
        // Fire post-relapse support immediately ΓÇö no throttle
        await StiraNotificationService.showNow(
          id: StiraNotificationService.ID_RELAPSE_SUPPORT,
          title: StiraNotificationCopy.postRelapse(s).title,
          body: StiraNotificationCopy.postRelapse(s).body,
          channel: StiraNotificationService.CHANNEL_CRITICAL,
          payload: "/home",
        );
        StiraThrottleManager.recordSent();
        break;
      case UserAction.vaultLetterSaved:
        // Cancel any pending high-intensity notification ΓÇö vault use counts as tool use
        await StiraNotificationService.cancel(StiraNotificationService.ID_URGE_VELOCITY);
        break;
      case UserAction.commitmentSaved:
        final commitment = Hive.box('commitments').values.last as Map?;
        final commitmentText = commitment?['commitment'] as String? ?? '';
        // Schedule same-time-tomorrow notification for 24h from now
        await StiraNotificationService.scheduleAt(
          id: StiraNotificationService.ID_SAME_TIME_TOMORROW,
          title: StiraNotificationCopy.sameTimeTomorrow(commitmentText, s.userName).title,
          body: StiraNotificationCopy.sameTimeTomorrow(commitmentText, s.userName).body,
          scheduledTime: DateTime.now().add(const Duration(hours: 24)),
          channel: StiraNotificationService.CHANNEL_HIGH,
          payload: "/checkin?source=commitment",
        );
        break;
      case UserAction.appOpened:
        // Cancel disengagement notifications ΓÇö user is back
        await StiraNotificationService.cancel(StiraNotificationService.ID_DISENGAGEMENT_48H);
        await StiraNotificationService.cancel(StiraNotificationService.ID_DISENGAGEMENT_72H);
        break;
    }
  }
  // ΓöÇΓöÇ INTERNAL HELPERS ΓöÇΓöÇ
  static bool _shouldFireUrgeVelocity(BehaviourSnapshot s) {
    // Velocity > 3 means urge rose by 3+ points across last 3 check-ins
    return s.urgeVelocity >= 3.0 && s.checkedInToday;
  }
  static bool _isApproachingPeakWindow(BehaviourSnapshot s, DateTime now) {
    final minutesToPeak = (s.peakHour * 60) - (now.hour * 60 + now.minute);
    // Fire 30-45 minutes before peak, with sufficient confidence
    return minutesToPeak >= 30 && minutesToPeak <= 45 && s.peakHourConfidence >= 50;
  }
  static bool _shouldFireCheckInReminder(BehaviourSnapshot s, DateTime now) {
    // Fire if: no check-in today AND within 1 hour of peak window AND not already disengaged
    if (s.checkedInToday) return false;
    if (s.hoursSinceLastCheckIn >= 48) return false; // disengagement handles this
    final minutesToPeak = ((s.peakHour - now.hour) * 60).abs();
    return minutesToPeak <= 60;
  }
  static Future<void> _send({
    required int id,
    required NotificationContent content,
    String channel = StiraNotificationService.CHANNEL_HIGH,
    bool isCrisis = false,
  }) async {
    if (!isCrisis && !StiraThrottleManager.canSend(isCrisis: false)) return;
    await StiraNotificationService.showNow(
      id: id,
      title: content.title,
      body: content.body,
      channel: channel,
      payload: content.payload,
    );
    StiraThrottleManager.recordSent();
  }
}

  static const String TASK_LOCATION = 'stira_location_check';
  static Future<void> initialize() async {
    // Check permission
    LocationPermission permission = await Geolocator.checkPermission();

    // Register periodic location check every 5 minutes
    await Workmanager().registerPeriodicTask(
      'stira_location_task',
      TASK_LOCATION,
      frequency: const Duration(minutes: 5),
      constraints: Constraints(
        networkType: NetworkType.not_required,
      ),
      existingWorkPolicy: ExistingWorkPolicy.keep,
    );
  }
  // ΓöÇΓöÇ Called from callbackDispatcher when task == TASK_LOCATION ΓöÇΓöÇ
  static Future<void> checkLocation() async {
    Position? position;
    try {
      position = await Geolocator.getCurrentPosition(
        desiredAccuracy: LocationAccuracy.medium,
        timeLimit: const Duration(seconds: 10),
      );
    } catch (e) {
      return; // Location not available ΓÇö skip this cycle
    }
    final zonesBox = Hive.box('danger_zones');
    final notifBox = Hive.box('notification_state');
    final zones = zonesBox.values.cast<Map>().toList();
    for (final zone in zones) {
      final double? lat = zone['lat'];
      final double? lng = zone['lng'];
      final double radius = (zone['radius_meters'] as num?)?.toDouble() ?? 100;
      final String zoneName = zone['name'] ?? 'This location';
      final bool isDanger = zone['is_danger'] ?? true;
      if (lat == null || lng == null) continue;
      final double distance = Geolocator.distanceBetween(
        position.latitude, position.longitude, lat, lng,
      );
      if (distance <= radius) {
        // Inside zone radius
        final String zoneKey = 'zone_last_fired_${zone['id'] ?? zoneName}';
        final String? lastFiredStr = notifBox.get(zoneKey);
        final DateTime? lastFired = lastFiredStr != null ? DateTime.tryParse(lastFiredStr) : null;
        // Cooldown: 6 hours per zone to prevent spam
        if (lastFired != null && DateTime.now().difference(lastFired).inHours < 6) continue;
        if (isDanger) {
          // 1. Fire SOS Heartbeat first (haptic)
          await StiraNotificationService.fireSOSHeartbeat();
          // 2. Wait 2 seconds, then fire push notification
          await Future.delayed(const Duration(seconds: 2));
          await StiraNotificationService.showNow(
            id: StiraNotificationService.ID_DANGER_ZONE,
            title: StiraNotificationCopy.dangerZone(zoneName).title,
            body: StiraNotificationCopy.dangerZone(zoneName).body,
            channel: StiraNotificationService.CHANNEL_CRITICAL,
            payload: "/tools",
          );
          // Record trigger count for this zone
          final int triggerCount = zone['trigger_count'] ?? 0;
          zonesBox.putAt(zones.indexOf(zone), {...zone, 'trigger_count': triggerCount + 1});
        } else {
          // Safe zone ΓÇö positive reinforcement
          await StiraNotificationService.showNow(
            id: StiraNotificationService.ID_SAFE_ZONE,
            title: StiraNotificationCopy.safeZone(zoneName).title,
            body: StiraNotificationCopy.safeZone(zoneName).body,
            channel: StiraNotificationService.CHANNEL_NORMAL,
            payload: "/home",
          );
        }
        // Record when this zone last fired
        notifBox.put(zoneKey, DateTime.now().toIso8601String());',
      }
    }
  }
}
